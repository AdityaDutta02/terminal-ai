# Waitlist System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Gate Terminal AI behind a waitlist page, collect signups, and let the superadmin launch the platform with one click — emailing all waitlisted users and granting them 10 credits.

**Architecture:** Next.js middleware reads a `platform.config` DB flag (30s TTL cache) and redirects all traffic to `/waitlist` unless the platform is live or the user has a `beta_access` cookie. A `?access=beta` param sets the cookie for testers. The `/waitlist` page collects emails, the admin panel gets a Launch button, and `lib/auth.ts` grants waitlist credits on signup for late joiners.

**Tech Stack:** Next.js 15 App Router, TypeScript strict, PostgreSQL (`platform` schema), Resend (via `lib/email.ts`), Vitest, Tailwind CSS, Instrument Serif + DM Sans fonts, shadcn/ui AlertDialog.

## Task 2: Cached Config Module

**Files:**
- Create: `platform/lib/waitlist-config.ts`

This module is imported by `middleware.ts`. It caches the `waitlist_mode` DB read for 30 seconds so middleware doesn't query the DB on every request.

- [ ] **Step 1: Create `platform/lib/waitlist-config.ts`**

```typescript
import { db } from './db'

let cached: { value: boolean; expiresAt: number } | null = null
const TTL_MS = 30_000

export async function isWaitlistMode(): Promise<boolean> {
  const now = Date.now()
  if (cached && now < cached.expiresAt) return cached.value

  try {
    const result = await db.query<{ value: string }>(
      `SELECT value FROM platform.config WHERE key = 'waitlist_mode' LIMIT 1`,
    )
    const value = result.rows[0]?.value !== 'false'
    cached = { value, expiresAt: now + TTL_MS }
    return value
  } catch {
    // On DB error, default to waitlist mode (safe fallback)
    return true
  }
}

export function invalidateWaitlistCache(): void {
  cached = null
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { isWaitlistMode } from '@/lib/waitlist-config'

const BETA_KEY = process.env.BETA_ACCESS_KEY ?? 'beta'
const COOKIE_NAME = 'beta_access'
const BYPASS_PREFIXES = ['/waitlist', '/api/waitlist', '/_next', '/favicon']

export async function middleware(request: NextRequest): Promise<NextResponse> {
  const { pathname, searchParams } = request.nextUrl

  // Always allow waitlist page, waitlist API, Next.js internals, favicon
  if (BYPASS_PREFIXES.some((prefix) => pathname.startsWith(prefix))) {
    return NextResponse.next()
  }

  // Allow auth routes so users can sign in/up
  if (pathname.startsWith('/api/auth') || pathname.startsWith('/(auth)')) {
    return NextResponse.next()
  }

  // Beta cookie already set → bypass
  if (request.cookies.get(COOKIE_NAME)?.value === '1') {
    return NextResponse.next()
  }

  // ?access=beta param → set cookie and redirect to same path without param
  if (searchParams.get('access') === BETA_KEY) {
    const url = request.nextUrl.clone()
    url.searchParams.delete('access')
    const response = NextResponse.redirect(url)
    response.cookies.set(COOKIE_NAME, '1', {
      httpOnly: true,
      sameSite: 'lax',
      secure: process.env.NODE_ENV === 'production',
      maxAge: 60 * 60 * 24 * 7, // 7 days
      path: '/',
    })
    return response
  }

  // Check DB flag (cached 30s)
  const waitlistActive = await isWaitlistMode()
  if (!waitlistActive) return NextResponse.next()

  // Redirect to waitlist
  const url = request.nextUrl.clone()
  url.pathname = '/waitlist'
  url.search = ''
  return NextResponse.redirect(url)
}

export const config = {
  matcher: ['/((?!_next/static|_next/image|favicon.png).*)'],
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { z } from 'zod'
import { db } from '@/lib/db'
import { logger } from '@/lib/logger'
import { sendWaitlistConfirmationEmail } from '@/lib/email'

const JoinSchema = z.object({
  email: z.string().email('Invalid email address'),
  name: z.string().max(100).optional(),
})

// Simple in-memory rate limiter: 5 requests per IP per hour
const rateLimitMap = new Map<string, { count: number; resetAt: number }>()

function checkRateLimit(ip: string): boolean {
  const now = Date.now()
  const entry = rateLimitMap.get(ip)
  if (!entry || now >= entry.resetAt) {
    rateLimitMap.set(ip, { count: 1, resetAt: now + 60 * 60 * 1000 })
    return true
  }
  if (entry.count >= 5) return false
  entry.count++
  return true
}

export async function POST(request: NextRequest): Promise<NextResponse> {
  const ip = request.headers.get('x-forwarded-for')?.split(',')[0]?.trim() ?? 'unknown'

  if (!checkRateLimit(ip)) {
    return NextResponse.json({ error: 'Too many requests' }, { status: 429 })
  }

  let body: unknown
  try {
    body = await request.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON' }, { status: 400 })
  }

  const parsed = JoinSchema.safeParse(body)
  if (!parsed.success) {
    return NextResponse.json(
      { error: parsed.error.issues[0]?.message ?? 'Invalid input' },
      { status: 400 },
    )
  }

  const { email, name } = parsed.data

  const result = await db.query<{ id: string }>(
    `INSERT INTO platform.waitlist (email, name) VALUES ($1, $2)
     ON CONFLICT (email) DO NOTHING
     RETURNING id`,
    [email, name ?? null],
  )

  const isNew = (result.rowCount ?? 0) > 0
  if (isNew) {
    logger.info({ msg: 'waitlist_joined', email })
    await sendWaitlistConfirmationEmail(email)
  }

  return NextResponse.json({ joined: true })
}
```

```typescript
import { NextResponse } from 'next/server'
import { requireAdmin } from '@/lib/middleware/require-admin'
import { db } from '@/lib/db'
import { grantCredits } from '@/lib/credits'
import { logger } from '@/lib/logger'
import { sendWaitlistLaunchEmail } from '@/lib/email'
import { invalidateWaitlistCache } from '@/lib/waitlist-config'

type WaitlistRow = {
  email: string
  user_id: string | null
}

export async function POST(): Promise<NextResponse> {
  const auth = await requireAdmin()
  if (auth instanceof NextResponse) return auth

  // 1. Disable waitlist mode
  await db.query(
    `UPDATE platform.config SET value = 'false' WHERE key = 'waitlist_mode'`,
  )
  invalidateWaitlistCache()

  // 2. Fetch all waitlist entries joined against user accounts
  const waitlistResult = await db.query<WaitlistRow>(
    `SELECT w.email, u.id AS user_id
     FROM platform.waitlist w
     LEFT JOIN public."user" u ON u.email = w.email`,
  )
  const rows = waitlistResult.rows

  // 3. Grant 10 credits to users who already have accounts
  let creditsGranted = 0
  await Promise.all(
    rows
      .filter((row) => row.user_id !== null)
      .map(async (row) => {
        try {
          await grantCredits(row.user_id!, 10, 'waitlist_launch')
          creditsGranted++
        } catch (err) {
          logger.error({ msg: 'waitlist_credit_grant_failed', email: row.email, err: String(err) })
        }
      }),
  )

  // 4. Send launch emails in batches of 50
  const BATCH_SIZE = 50
  let emailsSent = 0
  for (let i = 0; i < rows.length; i += BATCH_SIZE) {
    const batch = rows.slice(i, i + BATCH_SIZE)
    await Promise.all(
      batch.map(async (row) => {
        try {
          await sendWaitlistLaunchEmail(row.email, row.user_id !== null)
          emailsSent++
        } catch (err) {
          logger.error({ msg: 'waitlist_launch_email_failed', email: row.email, err: String(err) })
        }
      }),
    )
  }

  // 5. Mark all as notified
  await db.query(
    `UPDATE platform.waitlist SET notified_at = NOW() WHERE notified_at IS NULL`,
  )

  logger.info({ msg: 'platform_launched', emailsSent, creditsGranted })

  return NextResponse.json({ launched: true, emailsSent, creditsGranted })
}
```
